def show_playlists(playlists):
    chocolate = True
    while chocolate:
        for i, playlist in enumerate(playlists['items']):
            if i % 2:

            else:
                print("%4d %32.32s" % (i + 1 + playlists['offset'], playlist['name']), end="\t")
        if playlists['next']:
            playlists = sp.next(playlists)
        else:
            chocolate = False
